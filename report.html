def performance_metric2(y_true, y_predict):
    """"""
    ss_res = np.sum((y_t - y_p)**2 for y_t, y_p in zip(y_true, y_predict))
    y_average = np.mean(y_true)
    ss_tot = np.sum((y - y_average)**2 for y in y_true)
    score = 1-ss_res/ss_tot

    return score

import numpy as np
import pandas as pd
from sklearn.cross_validation import train_test_split
from sklearn.metrics import make_scorer
from sklearn.model_selection import GridSearchCV, KFold
from sklearn.tree import DecisionTreeRegressor


data_bj = pd.read_csv('bj_housing.csv')
price = data_bj['Value']
feature = data_bj.drop('Value',axis=1)

feature_train, feature_test, price_train, price_test = train_test_split(feature, price, test_size=0.1, random_state=40)

def fit_models(X, y):
    """  [X,y]"""
    
    cross_validator = KFold()
    
    regressor = DecisionTreeRegressor()

    params = {'min_samples_split':np.arange(2,11),'max_depth': np.arange(1, 11, 1)}

    scoring_fnc = make_scorer(performance_metric)

    grid = GridSearchCV(regressor, params, scoring=scoring_fnc, cv=cross_validator)

    #  [X,y]
    grid = grid.fit(X, y)
    
    #print pd.DataFrame(grid.cv_results_)

    # 
    return grid.best_estimator_


def performance_metric2(y_true, y_predict):
    """"""
    from sklearn.metrics import r2_score
    score = r2_score(y_true, y_predict)

    return score

optimal_model = fit_models(feature_train, price_train)

print "Parameter 'max_depth' is {} for the optimal model.".format(optimal_model.get_params()['max_depth'])
print "Parameter 'min_samples_split' is {} for the optimal model.".format(optimal_model.get_params()['min_samples_split'])

pred = optimal_model.predict(feature_test)
r2 = performance_metric2(price_test,pred)
print "Optimal model has R^2 score {:,.2f} on test data".format(r2)

Parameter 'min_samples_split' is 4 for the optimal model.
